166 {
167  std::array<ngl::Vec3, 8> cascade;

169  //Distance from the look vector, on the right vector
170  float horizontal = tanf( Utility::radians(m_fov) / 1.0f );
171 
172  //float verticalFOV = m_fov / m_aspect;
173 
174  float verticalFOV = 2.0f * atan( tanf(Utility::radians(m_fov) / 2.0f) / m_aspect );
175 
176  float vertical = tanf(verticalFOV);
177 
178  _start = -_start;
179  _end = -_end;
180 
181  float sh = _start * horizontal;
182  float sv = _start * vertical;
183  float eh = _end * horizontal;
184  float ev = _end * vertical;
185 
186  //Near plane
187  cascade[0] = ngl::Vec3( -sh, sv, _start ); //Top left
188  cascade[1] = ngl::Vec3( sh, sv, _start ); //Top right
189  cascade[2] = ngl::Vec3( -sh, -sv, _start ); //Bottom left
190  cascade[3] = ngl::Vec3( sh, -sv, _start ); //Bottom right
191 
192  //Far plane
193  cascade[4] = ngl::Vec3( -eh, ev, _end ); //Top left
194  cascade[5] = ngl::Vec3( eh, ev, _end ); //Top right
195  cascade[6] = ngl::Vec3( -eh, -ev, _end ); //Bottom left.
196  cascade[7] = ngl::Vec3( eh, -ev, _end ); //Bottom right
197 
198  //Project from view space to world space
199  ngl::Mat4 iv = m_V;
200  //iv = iv.transpose();
201  iv = iv.inverse();
202 
203  for(auto &i : cascade)
204  {
205  ngl::Vec4 i4 = ngl::Vec4( i.m_x, i.m_y, i.m_z, 1.0f );
206  i4 = i4 * iv;
207  i = ngl::Vec3( i4.m_x, i4.m_y, i4.m_z );
208  }
209 
210  //Return cascade in world space.
211  return cascade;
212 }
