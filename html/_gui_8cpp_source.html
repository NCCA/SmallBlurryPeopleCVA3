411 void Gui::updateButtonArrays()
412 {
413  ngl::Vec2 res(m_win_w, m_win_h);
414  std::vector<ngl::Vec2> positions;
415  std::vector<ngl::Vec2> sizes;
416  std::vector<int> shader_ids;
417  std::vector<Action> actions;
418  int shader_id = 0;
419 
420  std::array<int, DOUBLE_MAX_NOTES> notification_ages{0};
421  size_t notification_ages_index = 0;
422  bool notification_uniform_needs_updating = false;
423  for(std::shared_ptr<Button> &button : m_buttons)
424  {
425  Button *b = button.get();
426  if(b)
427  {
428  b->updatePos(res);
429  positions.push_back(b->getPos());
430  sizes.push_back(b->getSize());
431  shader_ids.push_back(shader_id);
432  actions.push_back(b->getAction());
433  if(b->getAction() == Action::NOTIFY)
434  {
435  notification_uniform_needs_updating = true;
436  if(notification_ages_index < notification_ages.size())
437  {
438  notification_ages[notification_ages_index] = shader_id;
439  notification_ages[notification_ages_index+1] = ((NotificationButton *)b)->getAge();
440  notification_ages_index+=2;
441  }
442  }
443  }
444  shader_id++;
445  }
446 
447  if(notification_uniform_needs_updating)
448  {
449  glUniform1iv(glGetUniformLocation(ngl::ShaderLib::instance()->getProgramID(m_shader_name), "notification_ages"), DOUBLE_MAX_NOTES, &(notification_ages[0]));
450  glUniform1i(glGetUniformLocation(ngl::ShaderLib::instance()->getProgramID(m_shader_name), "max_notification_age"), MAX_AGE);
451  }
452 
453  glBindVertexArray(m_vao_id);
454 
455  glBindBuffer(GL_ARRAY_BUFFER, m_vbo_ids[0]);
456  glBufferData(GL_ARRAY_BUFFER, sizeof(ngl::Vec2) * positions.size(), &(positions[0]), GL_DYNAMIC_DRAW);
457  // now fix this to the attribute buffer 0
458  glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, 0, 0);
459 
460  glEnableVertexAttribArray(0);
461 
462  glBindBuffer(GL_ARRAY_BUFFER, m_vbo_ids[1]);
463  glBufferData(GL_ARRAY_BUFFER, sizeof(ngl::Vec2) * sizes.size(), &(sizes[0]), GL_DYNAMIC_DRAW);
464  // now fix this to the attribute buffer 1
465  glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, 0, 0);
466 
467  glEnableVertexAttribArray(1);
468 
469  glBindBuffer(GL_ARRAY_BUFFER, m_vbo_ids[2]);
470  glBufferData(GL_ARRAY_BUFFER, sizeof(GL_INT) * shader_ids.size(), &(shader_ids[0]), GL_DYNAMIC_DRAW);
471  // now fix this to the attribute buffer 2
472  glVertexAttribPointer(2, 1, GL_FLOAT, GL_FALSE, 0, 0);
473 
474  glEnableVertexAttribArray(2);
475 
476  glBindBuffer(GL_ARRAY_BUFFER, m_vbo_ids[3]);
477  glBufferData(GL_ARRAY_BUFFER, sizeof(GL_INT) * actions.size(), &(actions[0]), GL_DYNAMIC_DRAW);
478  // now fix this to the attribute buffer 2
479  glVertexAttribPointer(3, 1, GL_FLOAT, GL_FALSE, 0, 0);
480 
481  glEnableVertexAttribArray(3);
482 
483  glBindVertexArray(0);
484  m_text_outdated = true;

487 void Gui::updateNotifications()
488 {
489  // flag for whether there are any notifications to updated
490  bool uniform_needs_updating = false;
491  std::array<int, DOUBLE_MAX_NOTES> ages{0};
492  size_t i = 0;
493  int shader_id = 0;
494  std::vector< std::shared_ptr<Button> > buttons_to_remove;
495  for(std::shared_ptr<Button> &button : m_buttons)
496  {
497  Button *b = button.get();
498  if(b->getAction() == Action::NOTIFY)
499  {
500  uniform_needs_updating = true;
501  ((NotificationButton *)b)->incrementAge();
502  if(((NotificationButton *)b)->getAge() > MAX_AGE)
503  {
504  buttons_to_remove.push_back(button);
505  }
506  else if(i < ages.size())
507  {
508  ages[i] = shader_id;
509  i++;
510  ages[i] = ((NotificationButton *)b)->getAge();
511  i++;
512  }
513  }
514  shader_id++;
515  }
516 
517  for(std::shared_ptr<Button> &button : buttons_to_remove)
518  {
519  removeButton(button);
520  }
521 
522  if(uniform_needs_updating)
523  {
524  glUniform1iv(glGetUniformLocation(ngl::ShaderLib::instance()->getProgramID(m_shader_name), "notification_ages"), DOUBLE_MAX_NOTES, &(ages[0]));
525  glUniform1i(glGetUniformLocation(ngl::ShaderLib::instance()->getProgramID(m_shader_name), "max_notification_age"), MAX_AGE);
526  }
