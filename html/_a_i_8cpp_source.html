22 bool AI::move()
23 {
24  // check whether the next point is necessary or if character has line of sight to the second point in list
25  if(m_path.size() > 1 && NodeNetwork::raytrace(m_grid, m_pos, m_path.end()[-2]))
26  {
27  // if character can see point after next, remove the next point so character heads directly to the furtherst one they can see
28  // done on a one by one basis so it does all happen in the same frame as pathfinding, wouldn't be super slow that way but marginally better this way
29  m_path.pop_back();
30  }
31 
32  // move by distance up to speed
33  float dist_moved = 0;
34  while(m_path.size() > 0 && dist_moved < m_speed)
35  {
36  float dist = 1;
37  ngl::Vec2 aim_vec = calcAimVec(&dist);
38  if(dist < m_speed)
39  {
40  aim_vec *= dist;
41  m_path.pop_back();
42  }
43  else
44  {
45  aim_vec *= m_speed;
46  }
47  dist_moved += aim_vec.length();
48  m_pos += aim_vec;
49  }
50  updateRot();
51  if(m_path.size() <= 0)

54  }
55  else return false;
