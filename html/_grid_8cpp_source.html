150 float Grid::getInterpolatedHeight(float _x, float _y)
151 {
152  //clamping the input values to 0 and their respective limits
153  //handles edge cases for the map
154  float x0 = Utility::clamp(std::round(_x), 0, m_w-1);
155  float x1 = Utility::clamp(x0 + 1, 0, m_w);
156  float y0 = Utility::clamp(std::round(_y), 0, m_h-1);
157  float y1 = Utility::clamp(y0 + 1, 0, m_h);
158 
159  //retrieving the 4 height values to interpolate between
160  float h0 = m_tiles[x0 + m_w * y0].getHeight();
161  float h1 = m_tiles[x1 + m_w * y0].getHeight();
162  float h2 = m_tiles[x0 + m_w * y1].getHeight();
163  float h3 = m_tiles[x1 + m_w * y1].getHeight();
164 
165  //interpolate in the x direction
166  float h4 = h0 * (x1 - _x) + h1 * (_x - x0);
167  float h5 = h2 * (x1 - _x) + h3 * (_x - x0);
168 
169  //interpolate in the y direction
170  float h6 = h4 * (y1 - _y) + h5 * (_y - y0);
171 
172  return h6;
173 }
