21 
22  // global open flag = false
23  bool open_nodes_available = false;

25  for(Node &node : m_nodes)
26  {
27  // if open:
28  if(node.isOpen())
29  {
30  open_nodes_available = true;
31  // compare node cost to current lowest
32  if(lowest_f_node_id == -1)
33  {
34  lowest_f_node_id = node.getID();
35  }
36  else if (node.getFCost() < m_nodes[lowest_f_node_id].getFCost())
37  {
38  // update lowest cost node
39  lowest_f_node_id = node.getID();
40  }
41  }
42  }
43 
44  // if no nodes are open
45  if(!open_nodes_available)
46  {
47  break;
48  }
49  // GET NEIGHBOURS AND UPDATE COSTS
50 
51  // create neighbour list, (4)
52  std::array<int, 4> neighbour_ids = m_nodes[lowest_f_node_id].getNeighbours();
53  // check for neighbours
54  // place neighbours that are not yet in the vector
55  ngl::Vec2 current_pos = m_nodes[lowest_f_node_id].getPos();
56 
57  for(size_t i = 0; i < neighbour_ids.size(); i++)
58  {
59  // if neighbour is not found
60  if(neighbour_ids[i] == -1)
61  {
62  ngl::Vec2 tile_pos(current_pos);
63  switch(i)
64  {
65  case Neighbour::UP:
66  if(current_pos[1] < m_grid->getH() - 1)
67  {
68  tile_pos += ngl::Vec2(0, 1);
69  }
70  break;
71  case Neighbour::DOWN:
72  if(current_pos[1] > 0)
73  {
74  tile_pos += ngl::Vec2(0, -1);
75  }
76  break;
77  case Neighbour::LEFT:
78  if(current_pos[0] > 0)
79  {
80  tile_pos += ngl::Vec2(-1, 0);
81  }
82  break;
83  case Neighbour::RIGHT:
84  if(current_pos[0] < m_grid->getW() - 1)
85  {
86  tile_pos += ngl::Vec2(1, 0);
87  }
88  break;
89  default:
90  std::cerr << "tile id not found" << std::endl;
91  }
92  if(tile_pos != current_pos)
93  {
94  neighbour_ids[i] = m_nodes.size();
95  m_nodes.push_back(Node(m_grid, &m_nodes, tile_pos, m_target_pos, m_nodes[lowest_f_node_id].getID()));
96  }
97  }
98  }
99 
100  // for each neighbour:
101  for(int neighbour_id : neighbour_ids)
102  {
103  if(neighbour_id != -1)
104  {
105  Node *neighbour = &(m_nodes[neighbour_id]);
106  // if open:
107  if(neighbour->isOpen())
108  {
109  // check G cost
110  float potential_g_cost = neighbour->calcNewGCost(m_nodes[lowest_f_node_id].getID());
111  if(potential_g_cost < neighbour->getGCost())
112  {
113  // if lower, update parent to current node
114  neighbour->setParent(m_nodes[lowest_f_node_id].getID());
115  }
116  }
117  }
118  }
119 
120  // CLOSE NODE
121  // close current node
122  m_nodes[lowest_f_node_id].close();
123 
124  // if current is target node, path is found
125  if(m_nodes[lowest_f_node_id].getPos() == m_target_pos)
126  {
127  path_found = true;
128  return createFoundPath(m_nodes[lowest_f_node_id]);
129  }
130  }
131  // no path found if loop exits
132  std::vector<ngl::Vec2> no_path;
133  return no_path;
