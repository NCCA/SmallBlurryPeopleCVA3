352 std::string Prefs::getPrefValueString(const std::string &_key)
353 {
354  PrefType type = getTypeOfPref(_key);
355  std::string str_out;
356  std::stringstream stream;
357  int num_decimals = 3;
358  switch (type) {
359  case PrefType::BOOL:
360  str_out = boolToString(getBoolPref(_key));
361  break;
362  case PrefType::INT:
363  str_out = std::to_string(getIntPref(_key));
364  break;
365  case PrefType::FLOAT:
366  switch (getIncType(_key)) {
367  case IncType::NONE:
368  case IncType::X1:
369  case IncType::X10:
370  case IncType::X100:
371  case IncType::X1000:
372  num_decimals = 0;
373  break;
374  case IncType::X0_1:
375  num_decimals = 1;
376  break;
377  case IncType::X0_01:
378  num_decimals = 2;
379  break;
380  case IncType::X0_001:
381  num_decimals = 3;
382  break;
383  case IncType::X0_0001:
384  num_decimals = 4;
385  break;
386  case IncType::POW_2:
387  num_decimals = 3;
388  break;
389  default:
390  break;
391  }
392  stream << std::fixed << std::setprecision(num_decimals) << getFloatPref(_key);
393  str_out = stream.str();
394  break;
395  case PrefType::STRING:
396  str_out = getStrPref(_key);
397  break;
398  default:
399  std::cerr << "error in setPref, type not known" << std::endl;
400  str_out = "ERROR";
401  break;
402  }
403  return str_out;
404 }
