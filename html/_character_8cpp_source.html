1136 void Character::distanceSort(int io_left, int io_right, std::vector<ngl::Vec2> &_vector)
1137 {
1138  //quicksort algorithm
1139  int i = io_left, j = io_right;
1140  int pivot = (io_left + io_right) / 2;
1141 
1142  //find squared distance of pivot element of vector from character's position
1143  float pivot_dist = Utility::sqrDistance(_vector[pivot], m_pos);
1144 
1145  while (i <= j)
1146  {
1147  while (Utility::sqrDistance(_vector[i], m_pos) < pivot_dist) i++;
1148  while (Utility::sqrDistance(_vector[j], m_pos) > pivot_dist) j--;
1149 
1150  if (i <= j)
1151  {
1152  //swap elements in vector
1153  ngl::Vec2 tmpCoord = _vector[i];
1154  _vector[i] = _vector[j];
1155  _vector[j] = tmpCoord;
1156  i++;
1157  j--;
1158  }
1159  }
1160 
1161  //recursive call, subdividing the array for sorting
1162  if (io_left < j)
1163  distanceSort(io_left, j, _vector);
1164  if (i < io_right)
1165  distanceSort (i, io_right, _vector);
1166 }
